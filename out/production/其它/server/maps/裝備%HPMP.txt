    public int getItemIncMHPr(int itemId) {//裝備%HP
        if (getEquipStats(itemId) == null || !getEquipStats(itemId).containsKey("MHPr")) {
            return 0;
        }
        return getEquipStats(itemId).get("MHPr");
    }
